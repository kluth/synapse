Usage Example

<!-- Just use the custom elements in your HTML -->
<touch-receptor
  label="Click Me"
  variant="primary"
  size="medium"
></touch-receptor>

<text-receptor
  placeholder="Enter text..."
  type="text"
  required
></text-receptor>

<script>
// Listen for events
document.querySelector('touch-receptor')
  .addEventListener('touch', (e) => {
    console.log('Button touched!', e.detail);
  });

document.querySelector('text-receptor')
  .addEventListener('data-input', (e) => {
    console.log('Text entered:', e.detail.value);
  });
</script>
